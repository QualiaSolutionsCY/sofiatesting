    **1. Create the `telegram_group_messages` table via Supabase MCP:**

    Run SQL in Supabase Dashboard (via MCP execute_sql):
    ```sql
    CREATE TABLE IF NOT EXISTS telegram_group_messages (
      id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
      group_chat_id BIGINT NOT NULL,
      group_name TEXT,
      message_id BIGINT NOT NULL,
      sender_telegram_id BIGINT,
      sender_name TEXT,
      message_text TEXT,
      message_date TIMESTAMPTZ NOT NULL,
      created_at TIMESTAMPTZ NOT NULL DEFAULT NOW(),
      UNIQUE(group_chat_id, message_id)
    );

    -- Index for phone number text search
    CREATE INDEX idx_tgm_message_text ON telegram_group_messages USING gin(to_tsvector('english', message_text));
    -- Index for group lookup
    CREATE INDEX idx_tgm_group_chat_id ON telegram_group_messages(group_chat_id);
    -- Index for date-based cleanup
    CREATE INDEX idx_tgm_created_at ON telegram_group_messages(created_at);
    -- Retention: auto-delete messages older than 90 days (optional, via pg_cron later)
    ```

    Also enable RLS on the table (even though Edge Functions use service_role key):

    **2. Create `lib/telegram/message-indexer.ts`:**

    A simple function that stores group messages in the `telegram_group_messages` table. This function is called from the existing lead-router when any group message is received.

    ```typescript
    import "server-only";
    import { db } from "../db/client";
    // Add telegram_group_messages to the schema (see below)

    export async function indexGroupMessage(message: TelegramMessage): Promise<void> {
      // Only index messages with text content
      const text = message.text || message.caption;
      if (!text) return;

      try {
        await db.insert(telegramGroupMessage).values({
          groupChatId: message.chat.id,
          groupName: message.chat.title || null,
          messageId: message.message_id,
          senderTelegramId: message.from?.id || null,
          senderName: message.from ? `${message.from.first_name} ${message.from.last_name || ''}`.trim() : null,
          messageText: text.substring(0, 5000), // Cap at 5000 chars
          messageDate: new Date(message.date * 1000),
        }).onConflictDoNothing(); // Skip if already indexed (duplicate webhook)
      } catch (error) {
        // Non-critical - don't fail message processing over indexing
        console.error('Error indexing group message:', error);
      }
    }
    ```
