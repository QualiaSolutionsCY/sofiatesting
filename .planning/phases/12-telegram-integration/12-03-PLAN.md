    **1. Create `lib/telegram/audit-response-handler.ts`:**

    A Next.js-compatible wrapper that handles audit alert responses. This duplicates the core logic from the Deno module but uses the Next.js Supabase client (from `lib/db/client.ts` or direct `@supabase/supabase-js`).

    ```typescript
    import "server-only";
    import { createClient } from "@supabase/supabase-js";
    import { logger } from "../logger";
    import type { TelegramMessage } from "./types";

    const log = logger.telegram.child("audit-response");

    // Vasya's Telegram user ID (same as in _shared/telegram-search.ts)
    // TODO: Replace with actual ID from Fawzi
    const VASYA_TELEGRAM_USER_ID = 0;

    // Response patterns (duplicated from Deno module for Node.js compatibility)
    const FOUND_PATTERNS = /\b(found|yes|done|ok|handled|resolved|attended|got\s?it)\b/i;
    const NOT_FOUND_PATTERNS = /\b(not\s+found|no|can'?t\s+find|cannot\s+find|nothing|nobody)\b/i;
    const PHONE_PATTERN = /(?:\+?(?:00)?357)?[\s-]?(\d{2})[\s-]?(\d{3,4})[\s-]?(\d{3,4})/;

    type ResponseType = 'found' | 'not_found' | 'alternative_number' | 'unknown';

    function parseResponse(text: string): { type: ResponseType; alternativeNumber?: string } {
      const phoneMatch = text.match(PHONE_PATTERN);
      if (phoneMatch) {
        return { type: 'alternative_number', alternativeNumber: phoneMatch[0].replace(/[\s-]/g, '') };
      }
      if (FOUND_PATTERNS.test(text)) return { type: 'found' };
      if (NOT_FOUND_PATTERNS.test(text)) return { type: 'not_found' };
      return { type: 'unknown' };
    }
    ```
