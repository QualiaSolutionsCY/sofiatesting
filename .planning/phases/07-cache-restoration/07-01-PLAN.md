2. Create function to get current database version:
```typescript
async function getDatabaseVersion(supabase: SupabaseClient): Promise<string | null> {
  try {
    const { data, error } = await supabase
      .from("sophia_prompts")
      .select("updated_at")
      .eq("is_active", true)
      .order("updated_at", { ascending: false })
      .limit(1)
      .single();

    if (error || !data) {
      return null;
    }
    return data.updated_at;
  } catch {
    return null;
  }
}
```

3. Update `getPromptSections` to check version before using cache:
```typescript
async function getPromptSections(supabase: SupabaseClient): Promise<Map<string, string>> {
  const now = Date.now();

  // Check if cache exists and is within TTL
  if (cachedPromptSections && now - cacheTimestamp < CACHE_TTL_MS) {
    // Version check: verify DB hasn't been updated
    const dbVersion = await getDatabaseVersion(supabase);
    if (dbVersion && dbVersion === cacheVersion) {
      // Cache hit - version matches
      logger.debug("Cache hit - using cached prompts", {
        category: LogCategory.GENERAL,
        cacheAge: now - cacheTimestamp,
        version: cacheVersion,
      });
      return cachedPromptSections;
    } else {
      // Version mismatch - DB was updated
      logger.info("Cache version mismatch - refreshing", {
        category: LogCategory.GENERAL,
        cachedVersion: cacheVersion,
        dbVersion: dbVersion,
      });
    }
  }

  // Cache miss or stale - load from DB
  // ... existing load logic ...

  // After successful load, store the version
  const dbVersion = await getDatabaseVersion(supabase);
  cacheVersion = dbVersion;

  // ... rest of existing logic ...
}
```

4. Update `invalidateCache` to also clear version:
```typescript
export function invalidateCache(): void {

  cacheTimestamp = 0;
  cacheVersion = null;
  logger.info("Cache manually invalidated", { category: LogCategory.GENERAL });
}
```
