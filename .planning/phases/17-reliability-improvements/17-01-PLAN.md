```typescript
let loadingPromise: Promise<Map<string, string>> | null = null;

async function getPromptSections(supabase: SupabaseClient): Promise<Map<string, string>> {
  // Version/time-based cache hit path stays unchanged
  if (cachedPromptSections && !needsRefresh) {
    return cachedPromptSections;
  }

  // Cache miss - check if already loading
  if (loadingPromise) {
    logger.debug("Concurrent request detected - waiting for inflight load", { category: LogCategory.CACHE });
    return loadingPromise;
  }

  // Start loading - create promise BEFORE async work
  loadingPromise = (async () => {
    try {
      // Existing cache load logic here
      const merged = new Map(...);
      cachedPromptSections = merged;
      cacheTimestamp = Date.now();
      return merged;
    } finally {
      loadingPromise = null; // Clear when done
    }
  })();

  return loadingPromise;
}
```

**Key points:**
- Set loadingPromise BEFORE calling loadPromptSectionsFromDB (not after)
- Multiple concurrent calls share the same promise instance
- Clear loadingPromise in finally block (succeeds or fails)
- Preserve all existing cache logic (TTL, version checks, fallbacks)
- Add debug log when concurrent request detected
