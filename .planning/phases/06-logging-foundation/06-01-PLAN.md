```typescript
/**
 * Request Context Propagation
 * Enables correlation ID and context to flow through async call stacks
 */

export interface RequestContext {
  correlationId: string;
  startTime: number;
  category?: string;
  metadata?: Record<string, unknown>;
}

// Use AsyncLocalStorage for context propagation (Deno supports this)
const contextStorage = new AsyncLocalStorage<RequestContext>();

/**
 * Generate a unique correlation ID
 * Format: sophia-{timestamp}-{random}
 */
export function generateCorrelationId(): string {
  const timestamp = Date.now().toString(36);
  const random = Math.random().toString(36).substring(2, 8);
  return `sophia-${timestamp}-${random}`;
}

/**
 * Run a function with request context
 * All code within the callback has access to the context via getContext()
 */
export function withContext<T>(
  context: RequestContext,
  fn: () => T | Promise<T>
): T | Promise<T> {
  return contextStorage.run(context, fn);
}

/**
 * Get the current request context (returns undefined outside of withContext)
 */
export function getContext(): RequestContext | undefined {
  return contextStorage.getStore();
}

/**
 * Get correlation ID from context, or generate a new one if not in context
 */
export function getCorrelationId(): string {
  return getContext()?.correlationId ?? generateCorrelationId();
}
```

```typescript
export function categorizeError(error: Error | unknown): ErrorCategory {
  if (error instanceof Error) {
    const msg = error.message.toLowerCase();
    if (msg.includes("network") || msg.includes("fetch") || msg.includes("timeout") || msg.includes("econnrefused")) {
      return ErrorCategory.NETWORK;
    }
    if (msg.includes("unauthorized") || msg.includes("forbidden") || msg.includes("401") || msg.includes("403")) {
      return ErrorCategory.AUTH;
    }
    if (msg.includes("invalid") || msg.includes("required") || msg.includes("validation")) {
      return ErrorCategory.VALIDATION;
    }
    if (msg.includes("openrouter") || msg.includes("gemini") || msg.includes("ai") || msg.includes("model")) {
      return ErrorCategory.AI;
    }
    if (msg.includes("zyprus") || msg.includes("wasend") || msg.includes("resend") || msg.includes("api")) {
      return ErrorCategory.EXTERNAL_API;
    }
  }
  return ErrorCategory.UNKNOWN;
}
```
