This table is written by Edge Functions (sophia-bot) and queried for phone number searches. Structure from migration:
- group_chat_id (bigint)
- group_name (text)
- message_id (bigint)
- sender_telegram_id (bigint)

- message_date (timestamp)
- created_at (timestamp)

```sql
-- Edge Functions (service_role) can insert/select (bypasses RLS anyway)
-- Admin users can query for phone number searches
CREATE POLICY "Admin users can view telegram group messages"
  ON telegram_group_messages FOR SELECT
  USING (
    auth.uid() IN (
      SELECT id FROM "User"
      WHERE email IN (
        SELECT email FROM admin_users WHERE is_active = true
      )
    )
  );

CREATE POLICY "System can insert telegram group messages"
  ON telegram_group_messages FOR INSERT
  WITH CHECK (true);  -- Edge Functions use service_role, this just makes it explicit

-- No UPDATE/DELETE - append-only message index
```
