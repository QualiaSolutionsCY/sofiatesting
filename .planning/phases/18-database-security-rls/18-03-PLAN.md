**Policy pattern:**

For Phase 18, create restrictive policies that will be enhanced in Phase 19 with proper admin role checks:

```sql
-- Enable RLS
ALTER TABLE "AdminAuditLog" ENABLE ROW LEVEL SECURITY;
ALTER TABLE "admin_users" ENABLE ROW LEVEL SECURITY;
ALTER TABLE "AgentExecutionLog" ENABLE ROW LEVEL SECURITY;

-- AdminAuditLog - Placeholder admin policies
-- Note: Phase 19 will add proper admin role checks
CREATE POLICY "Admin users can view audit logs"
  ON "AdminAuditLog" FOR SELECT
  USING (
    -- Temporary: Check if user exists in admin_users table
    -- Phase 19 will replace with proper role-based check from auth.jwt()
    auth.uid() IN (
      SELECT id FROM "User"
      WHERE email IN (
        SELECT email FROM admin_users WHERE is_active = true
      )
    )
  );

CREATE POLICY "Admin users can insert audit logs"
  ON "AdminAuditLog" FOR INSERT
  WITH CHECK (
    auth.uid() IN (
      SELECT id FROM "User"
      WHERE email IN (
        SELECT email FROM admin_users WHERE is_active = true
      )
    )
  );

-- admin_users table
CREATE POLICY "Admin users can view admin roles"
  ON "admin_users" FOR SELECT
  USING (
    -- Self-access: Users can view their own admin record
    email = (SELECT email FROM "User" WHERE id = auth.uid())
    OR
    -- Admin access: Check via admin_users table
    (SELECT email FROM "User" WHERE id = auth.uid()) IN (
      SELECT email FROM admin_users
      WHERE is_active = true
      AND role IN ('superadmin', 'admin')
    )
  );

CREATE POLICY "Superadmin can modify admin roles"
  ON "admin_users" FOR ALL
  USING (
    (SELECT email FROM "User" WHERE id = auth.uid()) IN (
      SELECT email FROM admin_users
      WHERE is_active = true
      AND role = 'superadmin'
    )
  );

-- AgentExecutionLog
CREATE POLICY "Admin users can view execution logs"
  ON "AgentExecutionLog" FOR SELECT
  USING (
    -- Users can view their own execution logs
    userId = auth.uid()
    OR
    -- Admins can view all execution logs
    auth.uid() IN (
      SELECT id FROM "User"
      WHERE email IN (
        SELECT email FROM admin_users WHERE is_active = true
      )
    )
  );

CREATE POLICY "System can insert execution logs"
  ON "AgentExecutionLog" FOR INSERT
  WITH CHECK (true);  -- System-level inserts allowed

-- No UPDATE/DELETE policies - these are append-only audit tables
```

**Important notes:**
- These policies provide baseline protection for Phase 18
- Phase 19 will enhance with proper JWT role claims (auth.jwt() -> 'role' = 'admin')
- Edge Functions (service_role key) bypass RLS and can always write logs
- AdminAuditLog and AgentExecutionLog are append-only (no UPDATE/DELETE policies)
- admin_users requires superadmin role for modifications

**Migration structure:**

- Enable RLS on all 3 tables
