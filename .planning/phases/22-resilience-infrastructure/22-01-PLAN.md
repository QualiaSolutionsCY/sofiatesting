**Pattern to replicate:**
```typescript
import { canRequest, recordSuccess, recordFailure } from "../utils/circuit-breaker.ts";

const BREAKER_CONFIG = {
  name: "service-name",
  failureThreshold: 3,
  resetTimeoutMs: 60_000, // 1 minute
};

// Before fetch:
if (!canRequest(BREAKER_CONFIG)) {
  throw new Error("[ServiceName] Circuit breaker OPEN - failing fast");
}

// In try block after successful fetch:
recordSuccess(BREAKER_CONFIG);

// In catch block:
recordFailure(BREAKER_CONFIG);
throw error;
```
