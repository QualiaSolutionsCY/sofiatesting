Create supabase-server.ts module for RLS-aware operations (uses anon key instead of service role):

**Create singleton pattern:**
```typescript
import { createClient, type SupabaseClient } from "https://esm.sh/@supabase/supabase-js@2.39.0";

// Singleton server client (anon key - respects RLS)
let _serverClient: SupabaseClient | null = null;

/**
 * Get the shared Supabase server client (singleton)
 * Uses anon key - respects Row Level Security policies
 */
export const getSupabaseServer = (): SupabaseClient => {
  if (!_serverClient) {
    const supabaseUrl = Deno.env.get("SUPABASE_URL");
    const supabaseAnonKey = Deno.env.get("SUPABASE_ANON_KEY");

    if (!supabaseUrl || !supabaseAnonKey) {
      throw new Error("Missing SUPABASE_URL or SUPABASE_ANON_KEY");
    }

    _serverClient = createClient(supabaseUrl, supabaseAnonKey);
  }
  return _serverClient;
};
```

**Note:** Currently all Edge Functions use service role key (admin client). This creates the server client for future RLS-aware operations. For this phase, focus on consolidating existing service role usages to getSupabaseAdmin().
